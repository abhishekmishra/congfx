481int compare_cells(cg_cell_t *cell1, cg_cell_t *cell2)
482{
483 if (cell1 == NULL && cell2 == NULL)
484 {
485 return 0;
486 }
487 if (cell1 == NULL || cell2 == NULL)
488 {
489 return -1;
490 }
491 if (cell1->c != cell2->c)
492 {
493 return -1;
494 }
495 if (cell1->bg.r != cell2->bg.r || cell1->bg.g != cell2->bg.g || cell1->bg.b != cell2->bg.b)
496 {
497 return -1;
498 }
499 if (cell1->fg.r != cell2->fg.r || cell1->fg.g != cell2->fg.g || cell1->fg.b != cell2->fg.b)
500 {
501 return -1;
502 }
503 return 0;
504}

530cg_canvas_t *make_canvas(cg_uint w, cg_uint h)
531{
532 cg_canvas_t *canvas = (cg_canvas_t *)_CG_CALLOC(1, sizeof(cg_canvas_t));
533 if (canvas == NULL)
534 {
535 wprintf(L"FATAL Error: Unable to allocate cg_canvas_t.\n");
536 exit(-1);
537 }
538 canvas->width = w;
539 canvas->height = h;
540 canvas->cells = (cg_cell_t *)_CG_CALLOC(w * h, sizeof(cg_cell_t));
541 if (canvas->cells == NULL)
542 {
543 wprintf(L"FATAL Error: Unable to allocate cg_cell_t array.\n");
544 exit(-1);
545 }
546
547 // allocate cells
548 for (cg_uint i = 0; i < h; i++)
549 {
550 for (cg_uint j = 0; j < w; j++)
551 {
552 canvas->cells[(i * w) + j] = *make_cell(_CG_DEFAULT_BACKGROUND_CHAR, default_bg_colour, default_fg_colour);
553 }
554 }
555 return canvas;
556}

617cg_uint _diff_time_micros(struct timespec time1, struct timespec time2)
618{
619 // wprintf(L"time1 [%ld, %ld], time2[%ld, %ld]\n", time1.tv_sec, time1.tv_nsec, time2.tv_sec, time2.tv_nsec);
620 cg_uint seconds_delta = (time1.tv_sec - time2.tv_sec);
621 cg_int nanos_delta = (time1.tv_nsec - time2.tv_nsec);
622 cg_uint delta = (seconds_delta * 1000000) + (nanos_delta / 1000);
623 // wprintf(L"seconds_delta = %lu, nanos_delta = %ld, delta = %lu\n", seconds_delta, nanos_delta, delta);
624 return delta;
625}

627int main(int argc, char *argv[])
628{
629 cg_char read_buf[20];
630
631 // enable raw mode for terminal
632 _cg_term_enable_raw_mode();
633
634 // init locale for terminal, and wide output
635 setlocale(LC_ALL, "");
636 fwide(stdout, 1);
637
638 // init random numbers
639 srand(time(NULL));
640
641 struct timespec start_time, prev_time, current_time, after_draw_time;
642 clock_gettime(CLOCK_MONOTONIC, &start_time);
643 prev_time = start_time;
644 current_time = start_time;
645
646 setup();
647
648 // if there is no canvas created, create a default one
649 if (canvas_current == NULL)
650 {
651 create_canvas(100, 25);
652 }
653
654 cg_uint delta_time_ideal = 1000000 / _fps; // in microseconds
655
656 while (_loop == 1)
657 {
658 cg_uint dt = _diff_time_micros(current_time, prev_time);
659
660 char c = '\0';
661 if (read(STDIN_FILENO, &c, 1) == -1 && errno != EAGAIN)
662 {
663 cg_err_fatal_msg(L"read");
664 }
665 // if (iscntrl(c))
666 // {
667 // wprintf(L"%d\r\n", c);
668 // }
669 // else
670 // {
671 // wprintf(L"%d ('%c')\r\n", c, c);
672 // }
673 if (c == 'q')
674 {
675 break;
676 }
677 key_pressed(c);
678
679 // no_loop();
680
681 // set default background and forground
682 _cg_term_reset();
683 _cg_term_set_foreground_colour(default_fg_colour);
684 background(default_bg_colour);
685 set_colour(default_fg_colour);
686
687 cls();
688 home();
689
690 draw(dt);
691 show_canvas();
692
693 // how much time spent
694 clock_gettime(CLOCK_MONOTONIC, &after_draw_time);
695 cg_uint dt_done = _diff_time_micros(after_draw_time, current_time);
696 // wprintf(L"Delta ideal %lu, Delta done %lu\n", delta_time_ideal, dt_done);
697 if (delta_time_ideal > dt_done)
698 {
699 struct timespec dt_diff, dt_diff_rem;
700 dt_diff.tv_sec = 0;
701 dt_diff.tv_nsec = (delta_time_ideal - dt_done) * 1000;
702 // sleep for the difference
703 // wprintf(L"sleep for -> [%ld]nanos\n", dt_diff.tv_nsec);
704 nanosleep(&dt_diff, &dt_diff_rem);
705 }
706
707 prev_time = current_time;
708 clock_gettime(CLOCK_MONOTONIC, &current_time);
709 // dt_done = _diff_time_micros(current_time, prev_time);
710 // wprintf(L"After sleep: Delta ideal %lu, Delta done %lu\n", delta_time_ideal, dt_done);
711
712 // swap canvas
713 swap_canvas();
714 }
715}

850void _cg_term_enable_raw_mode()
851{
852 if (tcgetattr(STDIN_FILENO, &orig_termios) == -1)
853 {
854 cg_err_fatal_msg(L"tcgetattr");
855 }
856
857 atexit(_cg_term_disable_raw_mode);
858 struct termios raw = orig_termios;
859 raw.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
860 raw.c_oflag &= ~(OPOST);
861 raw.c_cflag |= (CS8);
862 raw.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
863
864 raw.c_cc[VMIN] = 0;
865 raw.c_cc[VTIME] = 1;
866
867 if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw) == -1)
868 {
869 cg_err_fatal_msg(L"tcsetattr");
870 }
871
872 // Get the current flags
873 if ((_cg_term_orig_flags = fcntl(STDIN_FILENO, F_GETFL, 0)) == -1)
874 {
875 cg_err_fatal_msg(L"fcntl error getting flags");
876 }
877
878 // Set the flags to be non-blocking
879 if ((fcntl(STDIN_FILENO, F_SETFL, _cg_term_orig_flags | O_NONBLOCK) == -1))
880 {
881 cg_err_fatal_msg(L"fcntl error setting flags");
882 }
883}
884
885void _cg_term_disable_raw_mode()
886{
887 if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios) == -1)
888 {
889 cg_err_fatal_msg(L"tcsetattr");
890 }
891
892 // Reset the flags
893 if (fcntl(STDIN_FILENO, F_SETFL, _cg_term_orig_flags) == -1)
894 {
895 cg_err_fatal_msg(L"fcntl error resetting flags");
896 }
897}

969void show_canvas()
970{
971 if (canvas_current != NULL)
972 {
973 cg_rgb_t current_fg = {255, 255, 255};
974 cg_rgb_t current_bg = {0, 0, 0};
975 // wprintf(L"%ls", canvas_contents);
976 cg_uint idx = 0;
977 for (cg_uint i = 0; i < canvas_current->height; i++)
978 {
979 for (cg_uint j = 0; j < canvas_current->width; j++)
980 {
981 cg_cell_t *current_cell = get_cell(canvas_current, j, i);
982 cg_cell_t *previous_cell = get_cell(canvas_previous, j, i);
983 // if (compare_cells(current_cell, previous_cell) == 0)
984 // {
985 // continue;
986 // }
987 cg_rgb_t cell_fg = get_cell_fg(current_cell);
988 cg_rgb_t cell_bg = get_cell_bg(current_cell);
989
990 if (cell_fg.r != current_fg.r || cell_fg.g != current_fg.g || cell_fg.b != current_fg.b)
991 {
992 _cg_term_set_foreground_colour(cell_fg);
993 current_fg = cell_fg;
994 }
995 if (cell_bg.r != current_bg.r || cell_bg.g != current_bg.g || cell_bg.b != current_bg.b)
996 {
997 _cg_term_set_background_colour(cell_bg);
998 current_bg = cell_bg;
999 }
1000 // _cg_term_set_foreground_colour(canvas_foreground_colour[(i * width) + j]);
