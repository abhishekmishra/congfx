44/*
45********************************************************************************
46********************************** README **************************************
47********************************************************************************
48ConGfx - A simple graphics exploration library for the terminal.
49
50This graphics exploration library is inspired by p5.js, love2d, and the kilo
51text editor and corresponding tutorial to build an editor for the terminal from
52scratch.
53
54I've noticed that terminals over time have become quite capable, and looking at
55the cool things being built for the terminal I wondered how far I could go in
56building p5.js or love2d like game/graphics loop. Can we draw stuff on the
57terminal and refresh the drawing fast enough to create the illusion of moving
58pictures.
59
60********************************************************************************
61********************************************************************************
62*/

862int cg_compare_cells(cg_cell_t *cell1, cg_cell_t *cell2)
863{
864 if (cell1 == NULL && cell2 == NULL)
865 {
866 return 0;
867 }
868 if (cell1 == NULL || cell2 == NULL)
869 {
870 return -1;
871 }
872 if (cell1->c != cell2->c)
873 {
874 return -1;
875 }
876 if (cell1->bg.r != cell2->bg.r || cell1->bg.g != cell2->bg.g || cell1->bg.b != cell2->bg.b)
877 {
878 return -1;
879 }
880 if (cell1->fg.r != cell2->fg.r || cell1->fg.g != cell2->fg.g || cell1->fg.b != cell2->fg.b)
881 {
882 return -1;
883 }
884 return 0;

887cg_canvas_t *cg_make_canvas(cg_uint w, cg_uint h)
888{
889 cg_canvas_t *canvas = (cg_canvas_t *)_CG_CALLOC(1, sizeof(cg_canvas_t));
890 if (canvas == NULL)
891 {
892 printf("FATAL Error: Unable to allocate cg_canvas_t.\n");
893 exit(-1);
894 }
895 canvas->width = w;
896 canvas->height = h;
897 canvas->cells = (cg_cell_t *)_CG_CALLOC(w * h, sizeof(cg_cell_t));
898 if (canvas->cells == NULL)
899 {
900 printf("FATAL Error: Unable to allocate cg_cell_t array.\n");
901 exit(-1);
902 }
903
904 // allocate cells
905 for (cg_uint i = 0; i < h; i++)
906 {
907 for (cg_uint j = 0; j < w; j++)
908 {
909 canvas->cells[(i * w) + j] = *cg_make_cell(_CG_DEFAULT_BACKGROUND_CHAR, default_bg_colour, default_fg_colour);
910 }
911 }
912 return canvas;
913}

939cg_uint _diff_time_micros(struct timespec time1, struct timespec time2)
940{
941 // printf("time1 [%ld, %ld], time2[%ld, %ld]\n", time1.tv_sec, time1.tv_nsec, time2.tv_sec, time2.tv_nsec);
942 cg_uint seconds_delta = (time1.tv_sec - time2.tv_sec);
943 cg_int nanos_delta = (time1.tv_nsec - time2.tv_nsec);
944 cg_uint delta = (seconds_delta * 1000000) + (nanos_delta / 1000);
945 // printf("seconds_delta = %lu, nanos_delta = %ld, delta = %lu\n", seconds_delta, nanos_delta, delta);
946 return delta;
947}
